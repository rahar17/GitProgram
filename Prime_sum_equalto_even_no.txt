// Apply Goldbach’s conjecture 

#include<bits/stdc++.h>

#define ll long long 

using namespace std;



bool isPrime(int a)
{

		
 for (int i = 2; i <=sqrt(a); i++)

 {
			
  if (a % i == 0)
		
   return false;
 }		
  return true;
 
}




vector<int> primesum(int A) 

{
    
 vector<int> v; 
    
   
 v.clear();
   
 
    
 for (int i = 2; i <= A / 2; i++) 

 {
			  
   if (isPrime(i) && isPrime(A - i)) 
			
   {
			   
     v.push_back(i);
           
     v.push_back(A-i);
	
   }

 }
    
   
 return v;

}

{
	
  int n;
    
  vector<int> vector;

  cin>>n;    // even no.

	

  vector=primesum(n);

  cout<<vector[i]<<" ";

}
